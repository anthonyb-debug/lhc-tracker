Setting Up the Apps Script Web App

This script acts as a bridge so the app can write disposition values back to your Google Sheet.

  1. Open your Google Sheet → Extensions → Apps Script
  2. Paste the code below into the editor and save (Ctrl+S)
  3. Click Deploy → New deployment → Web App
  4. Set Execute as: Me and Who has access: Anyone
  5. Click Deploy → copy the Web App URL → paste it in the field above
function doPost(e) {
  try {
    var params = JSON.parse(e.postData.contents);
    var sheet = SpreadsheetApp.getActiveSpreadsheet()
                  .getSheetByName(params.sheetName || "Sheet1");
    var row = params.row;     // row number (1-based)
    var col = params.col;     // column number (7 = G)
    var val = params.value;
    sheet.getRange(row, col).setValue(val);
    return ContentService
      .createTextOutput(JSON.stringify({success: true}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({success: false, error: err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet(e) {
  return ContentService
    .createTextOutput(JSON.stringify({status: "ok"}))
    .setMimeType(ContentService.MimeType.JSON);
}
